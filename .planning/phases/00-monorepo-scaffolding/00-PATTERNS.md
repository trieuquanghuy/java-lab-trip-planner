### `libs/observability/.../ReactiveMdcEnrichmentFilter.java` (notes for adaptation)

The servlet filter at `00-RESEARCH.md lines 1037-1083` is the template. For the WebFlux equivalent, implement `org.springframework.web.server.WebFilter` with:
- `@Override Mono<Void> filter(ServerWebExchange exchange, WebFilterChain chain)`
- Read `exchange.getRequest().getHeaders().getFirst("X-Request-Id")` instead of `req.getHeader(...)`
- Use Reactor context to propagate MDC: `chain.filter(exchange).contextWrite(...)`. The `Tracer` injection and `MDC.put(...)` calls remain identical.
- Register in `ObservabilityAutoConfiguration` under `@ConditionalOnClass(name = "org.springframework.web.server.WebFilter")`.

```java
// Source: 00-CONTEXT.md D-05, docs/04-api-spec.md §6
package com.tripplanner.errors;

import org.springframework.http.HttpStatus;
import org.springframework.http.ProblemDetail;

import java.net.URI;

public final class ProblemDetailFactory {

    private static final String TYPE_BASE = "https://tripplanner.example.com/errors/";

    private ProblemDetailFactory() {}

    public static ProblemDetail of(HttpStatus status, ErrorCode code, String detail) {
        ProblemDetail pd = ProblemDetail.forStatusAndDetail(status, detail);
        pd.setType(URI.create(TYPE_BASE + code.code()));
        pd.setProperty("code", code.code());
        return pd;
    }
}
```

```bash
#!/usr/bin/env bash
# Source: 00-CONTEXT.md D-33, ROADMAP.md Phase 0 SC #1-#5
set -euo pipefail

fail() { echo "[smoke] FAIL: $1" >&2; exit 1; }
pass() { echo "[smoke] OK:   $1"; }

# SC #1: compose healthy < 60s (caller is responsible for timing the docker compose up wrapper;
# this script verifies the steady-state).
docker compose ps --format json | jq -e 'all(.[]; .State == "running" and (.Health == "healthy" or .Health == ""))' \
  >/dev/null || fail "SC#1 not all services running+healthy"
pass "SC#1 compose healthy"

# SC #2: 4 services in Eureka (auth, trip, destination, api-gateway)
EUREKA="http://localhost:8761/eureka/apps"
for svc in AUTH-SERVICE TRIP-SERVICE DESTINATION-SERVICE API-GATEWAY; do
  curl -sf -H "Accept: application/json" "$EUREKA" | grep -q "\"name\":\"$svc\"" \
    || fail "SC#2 $svc not registered with Eureka"
done
pass "SC#2 4 services registered with Eureka"

# SC #3: gateway /actuator/health returns UP
curl -sf http://localhost:8080/actuator/health | grep -q '"status":"UP"' \
  || fail "SC#3 gateway /actuator/health not UP"
pass "SC#3 gateway /actuator/health UP"

# SC #4: frontend renders without console errors (smoke = HTTP 200 + non-empty body)
# Console-error check belongs in Playwright; Phase 0 only asserts HTTP reachability.
curl -sf -o /tmp/index.html http://localhost:5173 || fail "SC#4 frontend not reachable"
test -s /tmp/index.html || fail "SC#4 frontend body empty"
pass "SC#4 frontend reachable"

# SC #5: per-service flyway history tables present
for pair in "auth:auth_flyway_schema_history" "trip:trip_flyway_schema_history" "destination:destination_flyway_schema_history"; do
  schema="${pair%%:*}"; table="${pair##*:}"

    -tAc "SELECT 1 FROM information_schema.tables WHERE table_schema='$schema' AND table_name='$table'" \
    | grep -q '^1$' || fail "SC#5 $schema.$table missing"
done
pass "SC#5 all 3 per-service flyway history tables present"

echo "[smoke] ALL PASS"
```
