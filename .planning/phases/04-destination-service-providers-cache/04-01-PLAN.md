**DestinationsCacheRepository.java:**
```java
public interface DestinationsCacheRepository extends JpaRepository<DestinationsCacheEntity, String> {

    @Query(value = """
        SELECT d.* FROM destination.destinations_cache d
        WHERE earth_box(ll_to_earth(CAST(:lat AS float8), CAST(:lng AS float8)), :radius)
              @> ll_to_earth(CAST(d.lat AS float8), CAST(d.lng AS float8))
          AND earth_distance(
                ll_to_earth(CAST(:lat AS float8), CAST(:lng AS float8)),
                ll_to_earth(CAST(d.lat AS float8), CAST(d.lng AS float8))
              ) <= :radius
        ORDER BY earth_distance(
                   ll_to_earth(CAST(:lat AS float8), CAST(:lng AS float8)),
                   ll_to_earth(CAST(d.lat AS float8), CAST(d.lng AS float8))
                 ) ASC
        LIMIT :limit
        """, nativeQuery = true)
    List<DestinationsCacheEntity> findNearby(
        @Param("lat") double lat,
        @Param("lng") double lng,
        @Param("radius") double radius,
        @Param("limit") int limit);

    @Query(value = """
        SELECT d.* FROM destination.destinations_cache d
        WHERE d.fetched_at > NOW() - INTERVAL '24 hours'
          AND earth_box(ll_to_earth(CAST(:lat AS float8), CAST(:lng AS float8)), :radius)
              @> ll_to_earth(CAST(d.lat AS float8), CAST(d.lng AS float8))
          AND earth_distance(
                ll_to_earth(CAST(:lat AS float8), CAST(:lng AS float8)),
                ll_to_earth(CAST(d.lat AS float8), CAST(d.lng AS float8))
              ) <= :radius
        ORDER BY earth_distance(
                   ll_to_earth(CAST(:lat AS float8), CAST(:lng AS float8)),
                   ll_to_earth(CAST(d.lat AS float8), CAST(d.lng AS float8))
                 ) ASC
        LIMIT :limit
        """, nativeQuery = true)
    List<DestinationsCacheEntity> findNearbyFresh(
        @Param("lat") double lat,
        @Param("lng") double lng,
        @Param("radius") double radius,
        @Param("limit") int limit);

    @Modifying
    @Query(value = """
        INSERT INTO destination.destinations_cache
            (provider_ref, name, category, rating, lat, lng, address, photos, opening_hours, website, raw, fetched_at)
        VALUES (:#{#e.providerRef}, :#{#e.name}, :#{#e.category}, :#{#e.rating},
                :#{#e.lat}, :#{#e.lng}, :#{#e.address},
                CAST(:#{#e.photos} AS jsonb), CAST(:#{#e.openingHours} AS jsonb),
                :#{#e.website}, CAST(:#{#e.raw} AS jsonb), :#{#e.fetchedAt})
        ON CONFLICT (provider_ref) DO UPDATE SET
            name = EXCLUDED.name,
            category = EXCLUDED.category,
            rating = EXCLUDED.rating,
            lat = EXCLUDED.lat,
            lng = EXCLUDED.lng,
            address = EXCLUDED.address,
            photos = EXCLUDED.photos,
            opening_hours = EXCLUDED.opening_hours,
            website = EXCLUDED.website,
            raw = EXCLUDED.raw,
            fetched_at = EXCLUDED.fetched_at
        WHERE destination.destinations_cache.fetched_at < EXCLUDED.fetched_at
        """, nativeQuery = true)
    void upsert(@Param("e") DestinationsCacheEntity entity);
}
```

Note: The `upsert` uses SpEL `#{}` to extract entity fields into the native query. If SpEL on native queries is problematic, fall back to individual `@Param` values and a separate method signature.
