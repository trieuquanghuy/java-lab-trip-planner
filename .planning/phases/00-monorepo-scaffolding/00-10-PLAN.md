    ```yaml
    # frontend.yml — Phase 0 SKELETON CI per D-15.

    name: frontend
    on:
      push:
      pull_request:
        branches: [main]
    jobs:
      build:
        runs-on: ubuntu-24.04
        defaults:
          run:

          - uses: actions/checkout@v4

          - name: Lint

          - name: Test
            run: pnpm test --run
          - name: Build
            run: pnpm build
    ```
