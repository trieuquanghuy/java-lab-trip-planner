   ```tsx
   import { Navigate, Outlet, useLocation } from 'react-router-dom';
   import { useAuthStore } from './auth.store';
   import { Skeleton } from '@/components/ui/skeleton';

   export function ProtectedRoute() {
     const isAuthenticated = useAuthStore((s) => !!s.accessToken);
     const isInitializing = useAuthStore((s) => s.isInitializing);
     const location = useLocation();

     if (isInitializing) {
       return (
         <div className="flex-1 flex items-center justify-center">
           <Skeleton className="h-8 w-48" />
         </div>
       );
     }

     if (!isAuthenticated) {
       return <Navigate to={`/login?next=${encodeURIComponent(location.pathname)}`} replace />;
     }

     return <Outlet />;
   }
   ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Form input → API call | User-provided email/password sent to backend |
| URL params → UI state | ?status= and ?next= params influence navigation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-08 | S (Spoofing) | LoginPage | mitigate | Form validates email format before submission; server rejects invalid credentials with generic error |
| T-07-09 | T (Tampering) | ?next= param | mitigate | Only use pathname from next param (no full URLs); encode with encodeURIComponent; validate starts with "/" before navigating |
| T-07-10 | I (Info Disclosure) | Error messages | mitigate | Server returns generic "Email or password is incorrect" — no account enumeration (backend enforced) |
| T-07-11 | S (Spoofing) | VerifyEmailPage ?status= | accept | Status param only controls which message to display — no security-sensitive action taken client-side |
</threat_model>
