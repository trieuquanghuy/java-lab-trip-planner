Create `frontend/src/features/trips/DayTabs.tsx`:
```typescript
import { useState } from 'react';
import type { TripDay } from '@/types/trip';

interface Props {
  days: TripDay[];
  activeDay: string;
  onDayChange: (dayId: string) => void;
}

export function DayTabs({ days, activeDay, onDayChange }: Props) {
  return (
    <div className="flex overflow-x-auto gap-1 px-2 py-2 border-b bg-background/80 backdrop-blur-sm sticky top-16 z-30">
      {days.map((day, idx) => (
        <button
          key={day.id}
          onClick={() => onDayChange(day.id)}
          className={`flex-shrink-0 px-4 py-2 rounded-full text-sm font-medium transition-colors
            ${activeDay === day.id
              ? 'bg-primary text-primary-foreground'
              : 'bg-muted text-muted-foreground hover:bg-muted/80'
            }`}
          aria-selected={activeDay === day.id}
          role="tab"
        >
          Day {idx + 1}
        </button>
      ))}
    </div>
  );
}
```
- Horizontal scrollable tab bar
- Pill-style buttons with active state
- `role="tab"` + `aria-selected` for a11y
- Sticky below header on mobile
