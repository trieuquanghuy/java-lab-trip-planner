```typescript
import { useEffect, useRef, useMemo } from 'react';
import { MapContainer, TileLayer, Marker, Popup, useMap } from 'react-leaflet';
import L from 'leaflet';
import type { TripDay } from '@/types/trip';

// Fix Leaflet default marker icon issue with bundlers
delete (L.Icon.Default.prototype as any)._getIconUrl;
L.Icon.Default.mergeOptions({
  iconRetinaUrl: 'https://unpkg.com/leaflet@1.9.4/dist/images/marker-icon-2x.png',
  iconUrl: 'https://unpkg.com/leaflet@1.9.4/dist/images/marker-icon.png',
  shadowUrl: 'https://unpkg.com/leaflet@1.9.4/dist/images/marker-shadow.png',
});

interface MarkerData {
  id: string;
  name: string;
  lat: number;
  lng: number;
}

interface Props {
  markers: MarkerData[];
}

function FitBounds({ markers }: { markers: MarkerData[] }) {
  const map = useMap();
  const prevCoordsRef = useRef<string>('');

  useEffect(() => {
    if (markers.length === 0) return;
    const coordsKey = markers.map(m => `${m.lat},${m.lng}`).join('|');
    if (coordsKey === prevCoordsRef.current) return; // no change
    prevCoordsRef.current = coordsKey;

    const bounds = L.latLngBounds(markers.map(m => [m.lat, m.lng]));
    map.fitBounds(bounds, { padding: [50, 50], maxZoom: 14 });
  }, [markers, map]);

  return null;
}

export function TripMap({ markers }: Props) {
  if (markers.length === 0) {
    return (
      <div className="flex items-center justify-center h-full text-muted-foreground text-sm">
        Add destinations to see them on the map
      </div>
    );
  }

  const center = useMemo(() => {
    const avgLat = markers.reduce((s, m) => s + m.lat, 0) / markers.length;
    const avgLng = markers.reduce((s, m) => s + m.lng, 0) / markers.length;
    return [avgLat, avgLng] as [number, number];
  }, [markers]);

  return (
    <MapContainer center={center} zoom={12} className="h-full w-full rounded-lg" scrollWheelZoom>
      <TileLayer
        attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>'
        url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
      />
      {markers.map(marker => (
        <Marker key={marker.id} position={[marker.lat, marker.lng]}>
          <Popup>{marker.name}</Popup>
        </Marker>
      ))}
      <FitBounds markers={markers} />
    </MapContainer>
  );
}
```

Key decisions:
- `fitBounds` only fires on mount and when marker coordinates ACTUALLY change (prevCoordsRef comparison) — prevents map jerk per CONTEXT.md
- Uses OpenStreetMap tiles (free, no API key)
- Marker icon fix for Vite bundler (known Leaflet issue)
- Empty state when no markers
